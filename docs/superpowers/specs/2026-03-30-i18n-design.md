# 国际化 (i18n) 设计文档

**日期:** 2026-03-30
**作者:** Claude
**版本:** 1.1
**状态:** 设计阶段

---

## 1. 需求概述

为海创元 AI 教育网站添加中英文切换功能，支持所有静态文本的翻译。

### 需求明细

1. **语言检测:** 自动检测浏览器语言，默认中文
2. **语言持久化:** 使用 localStorage 保存用户选择
3. **翻译风格:** 本地化为主，符合英语用户阅读习惯
4. **URL 处理:** 路径保持不变，只切换页面内容
5. **数据范围:** 只翻译静态文本，数据库内容保持原样显示

### 当前状态

- Header 组件已有语言切换按钮，但仅设置本地状态，未生效
- 所有静态文本（导航、按钮、标题等）硬编码为中文
- 无 i18n 库依赖
- 项目中不存在 `contexts/` 和 `hooks/` 目录，需要新建

---

## 2. 架构设计

### 2.1 核心组件

```
┌─────────────────────────────────────────────────────────────┐
│                        layout.tsx                           │
│  ┌─────────────────────────────────────────────────────┐   │
│  │              LanguageProvider (Context)              │   │
│  │  ┌───────────────────────────────────────────────┐  │   │
│  │  │  语言状态 (lang) + 切换函数 (setLang)          │  │   │
│  │  │  初始化: useEffect 避免 hydration mismatch     │  │   │
│  │  └───────────────────────────────────────────────┘  │   │
│  └─────────────────────────────────────────────────────┘   │
│                           │                                 │
│                           ▼                                 │
│  ┌─────────────────────────────────────────────────────┐   │
│  │                   Header / Footer                    │   │
│  │  ┌───────────────────────────────────────────────┐  │   │
│  │  │         useTranslation() Hook                  │  │   │
│  │  │  → t(key) → 返回对应语言的文本                │  │   │
│  │  └───────────────────────────────────────────────┘  │   │
│  └─────────────────────────────────────────────────────┘   │
└─────────────────────────────────────────────────────────────┘
```

### 2.2 文件结构

```
src/
├── locales/
│   ├── zh.json          # 中文翻译字典
│   └── en.json          # 英文翻译字典
├── lib/
│   └── i18n.ts          # 语言检测、初始化工具函数
├── contexts/            # 新建目录
│   └── LanguageContext.tsx  # 语言状态管理 Context
├── hooks/               # 新建目录
│   └── useTranslation.ts   # 翻译 Hook
└── components/
    ├── Header.tsx       # 改造：使用 t()
    ├── Footer.tsx       # 改造：使用 t()
    └── ...
```

### 3.2 语言检测与持久化

**存储键:** `hcy-website-language`
**存储值:** `"zh"` | `"en"`

**初始化流程（避免 hydration mismatch）:**
```typescript
// lib/i18n.ts
export type Language = 'zh' | 'en';

export function getInitialLanguage(): Language {
  // 只在客户端执行
  if (typeof window === 'undefined') return 'zh';

  // 1. 尝试从 localStorage 读取
  const saved = localStorage.getItem('hcy-website-language');
  if (saved === 'zh' || saved === 'en') return saved;

  // 2. 检测浏览器语言
  const browserLang = navigator.language;
  if (browserLang.startsWith('en')) return 'en';

  // 3. 默认中文
  return 'zh';
}

export function setLanguage(lang: Language) {
  localStorage.setItem('hcy-website-language', lang);
  // 更新 HTML lang 属性
  document.documentElement.lang = lang === 'zh' ? 'zh-CN' : 'en';
}
```

### 3.3 Context API 设计（解决 hydration mismatch）

```typescript
// contexts/LanguageContext.tsx
"use client";

import { createContext, useContext, useState, useEffect, ReactNode } from 'react';
import { translations } from '@/locales/translations';
import type { Language } from '@/lib/i18n';

interface LanguageContextValue {
  lang: Language;
  setLang: (lang: Language) => void;
  t: (key: string) => string;
}

const LanguageContext = createContext<LanguageContextValue | null>(null);

export function LanguageProvider({ children }: { children: ReactNode }) {
  // 初始状态固定为 'zh'，避免 SSR 不匹配
  const [lang, setLangState] = useState<Language>('zh');
  const [mounted, setMounted] = useState(false);

  // 组件挂载后初始化语言
  useEffect(() => {
    const initLang = async () => {
      const { getInitialLanguage } = await import('@/lib/i18n');
      const initialLang = getInitialLanguage();
      setLangState(initialLang);
      setMounted(true);
    };
    initLang();
  }, []);

  // 切换语言
  const setLang = (newLang: Language) => {
    setLangState(newLang);
    import('@/lib/i18n').then(({ setLanguage }) => {
      setLanguage(newLang);
    });
  };

  // 翻译函数
  const t = (key: string): string => {
    const keys = key.split('.');
    let value: any = translations[lang];
    for (const k of keys) {
      value = value?.[k];
    }
    // 开发环境下警告缺失的翻译
    if (process.env.NODE_ENV === 'development' && !value) {
      console.warn(`[i18n] Missing translation key: ${key} for language: ${lang}`);
    }
    return value || key;
  };

  return (
    <LanguageContext.Provider value={{ lang, setLang, t }}>
      {children}
    </LanguageContext.Provider>
  );
}

export function useTranslation() {
  const context = useContext(LanguageContext);
  if (!context) {
    throw new Error('useTranslation must be used within LanguageProvider');
  }
  return context;
}
```
